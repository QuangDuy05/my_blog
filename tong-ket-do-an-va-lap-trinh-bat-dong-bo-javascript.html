Lời kết và Kỹ thuật Async/Await trong JavaScript

Để khép lại hành trình xây dựng blog này, chúng ta cùng nhìn qua cách JavaScript xử lý các tác vụ mạng một cách mượt mà thông qua Async/Await. Đây là cách giúp website của Duy tải dữ liệu từ bãi xe mà không làm đơ trang web của người dùng.

async function getParkingStats() {
    try {
        // Gọi API lấy dữ liệu thực tế từ máy chủ
        const response = await fetch('https://api.duyparking.vn/v1/stats');
        const data = await response.json();
        console.log("Dữ liệu bãi xe:", data);
        document.getElementById("status").innerText = "Còn trống: " + data.slots;
    } catch (error) {
        console.error("Lỗi kết nối mạng: ", error);
    }
}
